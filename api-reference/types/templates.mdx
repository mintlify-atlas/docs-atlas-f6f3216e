## ExerciseTemplate

An exercise template that can be used in workouts and routines.

```typescript
export interface ExerciseTemplate {
  id: string;
  title: string;
  type: string;
  primary_muscle_group: string;
  secondary_muscle_groups: string[];
  is_custom: boolean;
}
```

<ResponseField name="id" type="string" required>
  Unique identifier for the exercise template
</ResponseField>

<ResponseField name="title" type="string" required>
  Title/name of the exercise
</ResponseField>

<ResponseField name="type" type="string" required>
  Type of exercise (e.g., "weight_reps", "duration", "distance")
</ResponseField>

<ResponseField name="primary_muscle_group" type="string" required>
  Primary muscle group targeted by this exercise
</ResponseField>

<ResponseField name="secondary_muscle_groups" type="string[]" required>
  Array of secondary muscle groups targeted by this exercise
</ResponseField>

<ResponseField name="is_custom" type="boolean" required>
  Whether this is a custom exercise created by the user (true) or a Hevy default (false)
</ResponseField>

## PaginatedExerciseTemplates

Paginated response containing multiple exercise templates.

```typescript
export interface PaginatedExerciseTemplates {
  page: number;
  page_count: number;
  exercise_templates: ExerciseTemplate[];
}
```

<ResponseField name="page" type="number" required>
  Current page number (1-indexed)
</ResponseField>

<ResponseField name="page_count" type="number" required>
  Total number of pages available
</ResponseField>

<ResponseField name="exercise_templates" type="ExerciseTemplate[]" required>
  Array of exercise templates on this page
</ResponseField>

## RoutineFolder

A folder for organizing routines.

```typescript
export interface RoutineFolder {
  id: number;
  index: number;
  title: string;
  updated_at: string;
  created_at: string;
}
```

<ResponseField name="id" type="number" required>
  Unique identifier for the routine folder
</ResponseField>

<ResponseField name="index" type="number" required>
  Position/order of the folder (0-indexed)
</ResponseField>

<ResponseField name="title" type="string" required>
  Title/name of the folder
</ResponseField>

<ResponseField name="updated_at" type="string" required>
  ISO 8601 timestamp of last update
</ResponseField>

<ResponseField name="created_at" type="string" required>
  ISO 8601 timestamp of creation
</ResponseField>
